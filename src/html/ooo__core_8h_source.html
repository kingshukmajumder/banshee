43 /* 2-level branch predictor:
44  * - L1: Branch history shift registers (bshr): 2^NB entries, HB bits of history/entry, indexed by XOR'd PC
45  * - L2: Pattern history table (pht): 2^LB entries, 2-bit sat counters, indexed by XOR'd bshr contents
46  * NOTE: Assumes LB is in [NB, HB] range for XORing (e.g., HB = 18 and NB = 10, LB = 13 is OK)
47  */
48 template<uint32_t NB, uint32_t HB, uint32_t LB>
49 class BranchPredictorPAg {
50  private:
51  uint32_t bhsr[1 << NB];
52  uint8_t pht[1 << LB];
53 
54  public:
55  BranchPredictorPAg() {
56  uint32_t numBhsrs = 1 << NB;
57  uint32_t phtSize = 1 << LB;
58 
59  for (uint32_t i = 0; i < numBhsrs; i++) {
60  bhsr[i] = 0;
61  }
62  for (uint32_t i = 0; i < phtSize; i++) {
63  pht[i] = 1; // weak non-taken
64  }
65 
66  static_assert(LB <= HB, "Too many PHT entries");
67  static_assert(LB >= NB, "Too few PHT entries (you'll need more XOR'ing)");
68  }

70  // Predicts and updates; returns false if mispredicted
71  inline bool predict(Address branchPc, bool taken) {
72  uint32_t bhsrMask = (1 << NB) - 1;
73  uint32_t histMask = (1 << HB) - 1;
74  uint32_t phtMask = (1 << LB) - 1;
75 
76  // Predict
77  // uint32_t bhsrIdx = ((uint32_t)( branchPc ^ (branchPc >> NB) ^ (branchPc >> 2*NB) )) & bhsrMask;
78  uint32_t bhsrIdx = ((uint32_t)( branchPc >> 1)) & bhsrMask;
79  uint32_t phtIdx = bhsr[bhsrIdx];
80 
81  // Shift-XOR-mask to fit in PHT
82  phtIdx ^= (phtIdx & ~phtMask) >> (HB - LB); // take the [HB-1, LB] bits of bshr, XOR with [LB-1, ...] bits
83  phtIdx &= phtMask;
84 
85  // If uncommented, behaves like a global history predictor
86  // bhsrIdx = 0;
87  // phtIdx = (bhsr[bhsrIdx] ^ ((uint32_t)branchPc)) & phtMask;
88 
89  bool pred = pht[phtIdx] > 1;
90 
91  // info("BP Pred: 0x%lx bshr[%d]=%x taken=%d pht=%d pred=%d", branchPc, bhsrIdx, phtIdx, taken, pht[phtIdx], pred);
92 
93  // Update
94  pht[phtIdx] = taken? (pred? 3 : (pht[phtIdx]+1)) : (pred? (pht[phtIdx]-1) : 0); //2-bit saturating counter
95  bhsr[bhsrIdx] = ((bhsr[bhsrIdx] << 1) & histMask ) | (taken? 1: 0); //we apply phtMask here, dependence is further away
96 
97  // info("BP Update: newPht=%d newBshr=%x", pht[phtIdx], bhsr[bhsrIdx]);
98  return (taken == pred);
99  }
100 };

134  if (!extraSlots) {

151  inline void advancePos(uint64_t& curCycle) {
152  occupancy -= curWin[curPos].count;
153  curWin[curPos].set(0, 0);
154  curPos++;
155  curCycle++;
156 
157  if (curPos == H) { // rebase
158  // info("[%ld] Rebasing, curCycle=%ld", curCycle/H, curCycle);
159  std::swap(curWin, nextWin);
160  curPos = 0;
161  uint64_t nextWinHorizon = curCycle + 2*H; // first cycle out of range
162 
163  if (!ubWin.empty()) {
164  UBWinIterator it = ubWin.begin();
165  while (it != ubWin.end() && it->first < nextWinHorizon) {
166  uint32_t nextWinPos = it->first - H - curCycle;
167  assert_msg(nextWinPos < H, "WindowStructure: ubWin elem exceeds limit cycle=%ld curCycle=%ld nextWinPos=%d", it->first, curCycle, nextWinPos);
168  nextWin[nextWinPos] = it->second;
169  // info("Moved %d events from unbounded window, cycle %ld (%d cycles away)", it->second, it->first, it->first - curCycle);
170  it++;
171  }
172  ubWin.erase(ubWin.begin(), it);
173  }

207  template <bool touchOccupancy, bool recordPort>
208  void scheduleInternal(uint64_t& curCycle, uint64_t& schedCycle, uint8_t portMask) {
209  // If the window is full, advance curPos until it's not
210  while (touchOccupancy && occupancy == WSZ) {
211  advancePos(curCycle);
212  }
213 
214  uint32_t delay = (schedCycle > curCycle)? (schedCycle - curCycle) : 0;
215 
216  // Schedule, progressively increasing delay if we cannot find a slot
217  uint32_t curWinPos = curPos + delay;
218  while (curWinPos < H) {
219  if (trySchedule<touchOccupancy, recordPort>(curWin[curWinPos], portMask)) {
220  schedCycle = curCycle + (curWinPos - curPos);
221  break;
222  } else {
223  curWinPos++;
224  }
225  }
226  if (curWinPos >= H) {
227  uint32_t nextWinPos = curWinPos - H;
228  while (nextWinPos < H) {
229  if (trySchedule<touchOccupancy, recordPort>(nextWin[nextWinPos], portMask)) {
230  schedCycle = curCycle + (nextWinPos + H - curPos);
231  break;
232  } else {
233  nextWinPos++;
234  }
235  }
236  if (nextWinPos >= H) {
237  schedCycle = curCycle + (nextWinPos + H - curPos);
238  UBWinIterator it = ubWin.lower_bound(schedCycle);
239  while (true) {
240  if (it == ubWin.end()) {
241  WinCycle wc = {0, 0};
242  bool success = trySchedule<touchOccupancy, recordPort>(wc, portMask);
243  assert(success);
244  ubWin.insert(std::pair<uint64_t, WinCycle>(schedCycle, wc));
245  } else if (it->first != schedCycle) {
246  WinCycle wc = {0, 0};
247  bool success = trySchedule<touchOccupancy, recordPort>(wc, portMask);
248  assert(success);
249  ubWin.insert(it /*hint, makes insert faster*/, std::pair<uint64_t, WinCycle>(schedCycle, wc));
250  } else {
251  if (!trySchedule<touchOccupancy, recordPort>(it->second, portMask)) {
252  // Try next cycle
253  it++;
254  schedCycle++;
255  continue;
256  } // else scheduled correctly
257  }
258  break;
259  }
260  // info("Scheduled event in unbounded window, cycle %ld", schedCycle);
261  }
262  }
263  if (touchOccupancy) occupancy++;
264  }

266  template <bool touchOccupancy, bool recordPort>
267  inline uint8_t trySchedule(WinCycle& wc, uint8_t portMask) {
268  static_assert(!(recordPort && !touchOccupancy), "Can't have recordPort and !touchOccupancy");
269  if (touchOccupancy) {
270  uint8_t availMask = (~wc.occUnits) & portMask;
271  if (availMask) {
272  // info("PRE: occUnits=%x portMask=%x availMask=%x", wc.occUnits, portMask, availMask);
273  uint8_t firstAvail = __builtin_ffs(availMask) - 1;
274  // NOTE: This is not fair across ports. I tried round-robin scheduling, and there is no measurable difference
275  // (in our case, fairness comes from following program order)
276  if (recordPort) lastPort = firstAvail;
277  wc.occUnits |= 1 << firstAvail;
278  wc.count++;
279  // info("POST: occUnits=%x count=%x firstAvail=%d", wc.occUnits, wc.count, firstAvail);
280  }
281  return availMask;
282  } else {
283  // This is a shadow req, port has only 1 bit set
284  uint8_t availMask = (~wc.occUnits) & portMask;
285  wc.occUnits |= portMask; // or anyway, no conditionals
286  return availMask;
287  }
