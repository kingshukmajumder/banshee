213  uint32_t rank(const MemReq* req) {
214  //Choose part to evict from as a part with highest *proportional* diff between tgt and actual sizes (minimize/smooth transients); if all parts are within limits, evict from own
215  uint32_t victimPart = mapper->getPartition(*req);
216  double maxPartDiff = 0.0;
217  if (partInfo[victimPart].size == 0) maxPartDiff = -2.0; //force a > 0-size partition
218  for (uint32_t p = 0; p < partitions; p++) {
219  double diff = ((int32_t)partInfo[p].size - (int32_t)partInfo[p].targetSize)/((double)(partInfo[p].targetSize + 1));
220  //info("YYY %d %f", p, diff);
221  if (diff > maxPartDiff && partInfo[p].size > 0) {
222  maxPartDiff = diff;
223  victimPart = p;
224  }
225  }
226  //assert(maxPartDiff >= -1e-8, "Evicting from non-full line! diff=%f victimPart %d (sz %d tgt %d) origPart %d", ); //someone must be over...
227  if (maxPartDiff < -1e-8) {
228  warn("Evicting from non-full part! diff=%f victimPart %d (sz %ld tgt %ld) origPart %d",
229  maxPartDiff, victimPart, partInfo[victimPart].size, partInfo[victimPart].targetSize, mapper->getPartition(*req));
