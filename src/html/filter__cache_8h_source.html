36 /* Extends Cache with an L0 direct-mapped cache, optimized to hell for hits
37  *
38  * L1 lookups are dominated by several kinds of overhead (grab the cache locks,
39  * several virtual functions for the replacement policy, etc.). This
40  * specialization of Cache solves these issues by having a filter array that
41  * holds the most recently used line in each set. Accesses check the filter array,
42  * and then go through the normal access path. Because there is one line per set,
43  * it is fine to do this without grabbing a lock.
44  */

110  inline uint64_t load(Address vAddr, uint64_t curCycle) {
111  Address vLineAddr = vAddr >> lineBits;
112  uint32_t idx = vLineAddr & setMask;
113  uint64_t availCycle = filterArray[idx].availCycle; //read before, careful with ordering to avoid timing races
114  if (vLineAddr == filterArray[idx].rdAddr) {
115  fGETSHit++;
116  return MAX(curCycle, availCycle);
117  } else {
118  return replace(vLineAddr, idx, true, curCycle);
119  }

136  uint64_t replace(Address vLineAddr, uint32_t idx, bool isLoad, uint64_t curCycle) {
137  Address pLineAddr;
138  // page num = vLineAddr shifted by 6 bits. So it is shifted by 12 bits in total (4KB page size)
139  if (_enable_tlb) {
140  Address vpgnum = vLineAddr >> 6;
141  uint64_t pgnum;
142  futex_lock(&filterLock);
143  if (_tlb.find(vpgnum) == _tlb.end()) {
144  do {
145  int64_t rand;
146  lrand48_r(&_buffer, &rand);
147  pgnum = rand & 0x000fffffffffffff;
148  } while (_exist_pgnum.find(pgnum) != _exist_pgnum.end());
149  _tlb[vpgnum] = pgnum;
150  _exist_pgnum.insert( pgnum );
151  } else
152  pgnum = _tlb[vpgnum];
153  pLineAddr = procMask | (pgnum << 6) | (vLineAddr & 0x3f);
154  } else
155  pLineAddr = procMask | vLineAddr;
156  MESIState dummyState = MESIState::I;
157  MemReq req = {pLineAddr, isLoad? GETS : GETX, 0, &dummyState, curCycle, &filterLock, dummyState, srcId, reqFlags};
158  uint64_t respCycle = access(req);
159 
160  //Due to the way we do the locking, at this point the old address might be invalidated, but we have the new address guaranteed until we release the lock
161 
162  //Careful with this order
163  Address oldAddr = filterArray[idx].rdAddr;
164  filterArray[idx].wrAddr = isLoad? -1L : vLineAddr;
165  filterArray[idx].rdAddr = vLineAddr;
166 
167  //For LSU simulation purposes, loads bypass stores even to the same line if there is no conflict,
168  //(e.g., st to x, ld from x+8) and we implement store-load forwarding at the core.
169  //So if this is a load, it always sets availCycle; if it is a store hit, it doesn't
170  if (oldAddr != vLineAddr) filterArray[idx].availCycle = respCycle;
171 
172  futex_unlock(&filterLock);
173  return respCycle;
174  }

176  uint64_t invalidate(const InvReq& req) {
177  Cache::startInvalidate(); // grabs cache's downLock
178  futex_lock(&filterLock);
179  uint32_t idx = req.lineAddr & setMask; //works because of how virtual<->physical is done...
180  if ((filterArray[idx].rdAddr | procMask) == req.lineAddr) { //FIXME: If another process calls invalidate(), procMask will not match even though we may be doing a capacity-induced invalidation!
181  filterArray[idx].wrAddr = -1L;
182  filterArray[idx].rdAddr = -1L;
183  }
184  uint64_t respCycle = Cache::finishInvalidate(req); // releases cache's downLock
185  futex_unlock(&filterLock);
186  return respCycle;
187  }
