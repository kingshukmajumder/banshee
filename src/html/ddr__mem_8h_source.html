40 /* Efficiently track the activation window: A circular buffer that stores the
41  * next allowed cycle we're allowed to issue an activation
42  */
43 class ActWindow {
44  private:
45  g_vector<uint64_t> buf;
46  uint32_t idx;
47 
48  public:
49  void init(uint32_t size) {
50  buf.resize(size);
51  for (uint32_t i = 0; i < size; i++) buf[i] = 0;
52  idx = 0;
53  }
54 
55  inline uint64_t minActCycle() const {
56  return buf[idx];
57  }
58 
59  inline void addActivation(uint64_t actCycle) {
60  assert(buf[idx] <= actCycle); // o/w we have violated tTAW/tFAW and more...
61 
62  // We need to reorder rank ACT commands, which may happen somewhat out of order
63  // Typically, acts come in order or nearly in order, so doing this is pretty efficient
64  // (vs e.g. scanning all last few acts to figure out the minumum constraint)
65  uint32_t cur = idx;
66  while (buf[dec(cur)] > actCycle) {
67  buf[cur] = buf[dec(cur)];
68  cur = dec(cur);
69  if (cur == idx) break; // we're the oldest in the window
70  }
71  buf[cur] = actCycle;
72 
73  idx = inc(idx);
74  }
75 
76  private:
77  inline uint32_t inc(uint32_t i) const { return (i < buf.size()-1)? i+1 : 0; }
78  inline uint32_t dec(uint32_t i) const { return i? i-1 : buf.size()-1; }
79 };

106  assert(!full());
107  Node* n = freeList.back();
108  freeList.pop_back();

111  }
