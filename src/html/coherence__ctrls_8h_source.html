65 /* A MESI coherence controller is decoupled in two:
66  * - The BOTTOM controller, which deals with keeping coherence state with respect to the upper level and issues
67  * requests (accesses) to upper levels.
68  * - The TOP controller, which keeps state of lines w.r.t. lower levels of the hierarchy (e.g. sharer lists),
69  * and issues requests (invalidates) to lower levels.
70  * The naming scheme is PROTOCOL-CENTRIC, i.e. if you draw a multi-level hierarchy, between each pair of levels
71  * there is a top CC at the top and a bottom CC at the bottom. Unfortunately, if you look at the caches, the
72  * bottom CC is at the top is at the bottom. So the cache class may seem a bit weird at times, but the controller
73  * classes make more sense.
74  */

247 static inline bool CheckForMESIRace(AccessType& type, MESIState* state, MESIState initialState) {
248  //NOTE: THIS IS THE ONLY CODE THAT SHOULD DEAL WITH RACES. tcc, bcc et al should be written as if they were race-free.
249  bool skipAccess = false;
250  if (*state != initialState) {
251  //info("[%s] Race on line 0x%lx, %s by childId %d, was state %s, now %s", name.c_str(), lineAddr, accessTypeNames[type], childId, mesiStateNames[initialState], mesiStateNames[*state]);
252  //An intervening invalidate happened! Two types of races:
253  if (type == PUTS || type == PUTX) { //either it is a PUT...
254  //We want to get rid of this line
255  if (*state == I) {
256  //If it was already invalidated (INV), just skip access altogether, we're already done
257  skipAccess = true;
258  } else {
259  //We were downgraded (INVX), still need to do the PUT
260  assert(*state == S);
261  //If we wanted to do a PUTX, just change it to a PUTS b/c now the line is not exclusive anymore
262  if (type == PUTX) type = PUTS;
263  }
264  } else if (type == GETX) { //...or it is a GETX
265  //In this case, the line MUST have been in S and have been INValidated
266  assert(initialState == S);
267  assert(*state == I);
268  //Do nothing. This is still a valid GETX, only it is not an upgrade miss anymore
269  } else { //no GETSs can race with INVs, if we are doing a GETS it's because the line was invalid to begin with!
270  panic("Invalid true race happened (?)");
271  }
272  }
273  return skipAccess;
274 }

306  bool startAccess(MemReq& req) {
307  assert((req.type == GETS) || (req.type == GETX) || (req.type == PUTS) || (req.type == PUTX));
308 
309  /* Child should be locked when called. We do hand-over-hand locking when going
310  * down (which is why we require the lock), but not when going up, opening the
311  * child to invalidation races here to avoid deadlocks.
312  */
313  if (req.childLock) {
314  futex_unlock(req.childLock);
315  }
316 
317  tcc->lock(); //must lock tcc FIRST
318  bcc->lock();
319 
320  /* The situation is now stable, true race-wise. No one can touch the child state, because we hold
321  * both parent's locks. So, we first handle races, which may cause us to skip the access.
322  */
323  bool skipAccess = CheckForMESIRace(req.type /*may change*/, req.state, req.initialState);

347  uint64_t processAccess(const MemReq& req, int32_t lineId, uint64_t startCycle, uint64_t* getDoneCycle = nullptr) {
348  uint64_t respCycle = startCycle;
349  //Handle non-inclusive writebacks by bypassing
350  //NOTE: Most of the time, these are due to evictions, so the line is not there. But the second condition can trigger in NUCA-initiated
351  //invalidations. The alternative with this would be to capture these blocks, since we have space anyway. This is so rare is doesn't matter,
352  //but if we do proper NI/EX mid-level caches backed by directories, this may start becoming more common (and it is perfectly acceptable to
353  //upgrade without any interaction with the parent... the child had the permissions!)
354  if (lineId == -1 || (((req.type == PUTS) || (req.type == PUTX)) && !bcc->isValid(lineId))) { //can only be a non-inclusive wback
355  assert(nonInclusiveHack);
356  assert((req.type == PUTS) || (req.type == PUTX));
357  respCycle = bcc->processNonInclusiveWriteback(req.lineAddr, req.type, startCycle, req.state, req.srcId, req.flags);
358  } else {
359  //Prefetches are side requests and get handled a bit differently
360  bool isPrefetch = req.flags & MemReq::PREFETCH;
361  assert(!isPrefetch || req.type == GETS);
362  uint32_t flags = req.flags & ~MemReq::PREFETCH; //always clear PREFETCH, this flag cannot propagate up
363 
364  //if needed, fetch line or upgrade miss from upper level
365  respCycle = bcc->processAccess(req.lineAddr, lineId, req.type, startCycle, req.srcId, flags);
366  if (getDoneCycle) *getDoneCycle = respCycle;
367  if (!isPrefetch) { //prefetches only touch bcc; the demand request from the core will pull the line to lower level
368  //At this point, the line is in a good state w.r.t. upper levels
369  bool lowerLevelWriteback = false;
370  //change directory info, invalidate other children if needed, tell requester about its state
371  respCycle = tcc->processAccess(req.lineAddr, lineId, req.type, req.childId, bcc->isExclusive(lineId), req.state,
372  &lowerLevelWriteback, respCycle, req.srcId, flags);
373  if (lowerLevelWriteback) {
374  //Essentially, if tcc induced a writeback, bcc may need to do an E->M transition to reflect that the cache now has dirty data
375  bcc->processWritebackOnAccess(req.lineAddr, lineId, req.type);
376  }
377  }
378  }
379  return respCycle;
380  }
