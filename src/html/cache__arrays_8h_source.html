32 /* General interface of a cache array. The array is a fixed-size associative container that

34  * cache components store tag data in non-associative arrays indexed by line ID.
35  */

40 
41  /* Runs replacement scheme, returns tag ID of new pos and address of line to write back*/
42  virtual uint32_t preinsert(const Address lineAddr, const MemReq* req, Address* wbLineAddr) = 0;
43

45  * NOTE: This method is guaranteed to be called after preinsert, although

48  */
