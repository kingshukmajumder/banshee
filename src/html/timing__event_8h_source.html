230  inline void visitChildren(F f) {
231  if (numChildren == 0) return;
232  //info("visit %p nc %d", this, numChildren);
233  if (numChildren == 1) {
234  f(&child);
235  } else {
236  TimingEventBlock* curBlock = children;

239  for (uint32_t i = 0; i < TIMING_BLOCK_EVENTS; i++) {
240  //info("visit %p i %d %p", this, i, curBlock->events[i]);
241  if (!curBlock->events[i]) {break;}
242  //info("visit %p i %d %p PASS", this, i, curBlock->events[i]);
243  f(&(curBlock->events[i]));
244  visitedChildren++;
245  }
246  curBlock = curBlock->next;
247  }
248  //info("visit %p multi done", this);
