44 /* Adjusts period to fire on the first phase following the target. Sets exponentially decreasing periods,
45  * so even if maxRate is horribly overestimated, it should have a very small cost (but there's room for
46  * optimization if this becomes an issue).
47  */
48 template<typename G, typename F>
49 class AdaptiveEvent : public Event {
50  private:
51  G get;
52  F fire;
53  uint64_t target;
54  uint64_t maxRate;
55 
56  public:
57  AdaptiveEvent(G _get, F _fire, uint64_t _start, uint64_t _target, uint64_t _maxRate) : Event(0), get(_get), fire(_fire), target(_target), maxRate(_maxRate) {
58  assert(target >= _start);
59  period = (target - _start)/maxRate;
60  if (!period) period = 1;
61  }
62 
63  // This will fire a bunch of times, we adjust the period to get the exact phase
64  // Gets called from an arbitrary process, cannot touch any proc-local state (including FFI)
65  void callback() {
66  uint64_t cur = get();
67  if (cur >= target) {
68  assert(cur - target <= maxRate); //otherwise, maxRate was wrong...
69  fire();
70  period = 0; //event queue will dispose of us
71  } else {
72  period = (target - cur)/maxRate;
73  if (period == 0) period = 1;
74  }
75  }
76 };

85  private:
86  g_multimap<uint64_t, Event*> evMap;
87  lock_t qLock;
88 
89  public:
90  EventQueue() { futex_init(&qLock); }

92  void tick() {
93  futex_lock(&qLock);
94  uint64_t curPhase = zinfo->numPhases;
95  g_multimap<uint64_t, Event*>::iterator it = evMap.begin();
96  while (it != evMap.end() && it->first <= curPhase) {
97  if (unlikely(it->first != curPhase)) panic("First event should have ticked on phase %ld, this is %ld", it->first, curPhase);
98  //if (it->first != curPhase) warn("First event should have ticked on phase %ld, this is %ld", it->first, curPhase);
99  Event* ev = it->second;
100  evMap.erase(it);
101  ev->callback(); //NOTE: Callback cannot call insert(), will deadlock (could use recursive locks if needed)
102  if (ev->getPeriod()) {
103  evMap.insert(std::pair<uint64_t, Event*>(curPhase + ev->getPeriod(), ev));
104  } else {
105  delete ev;
106  }
107  it = evMap.begin();
108  }
109  futex_unlock(&qLock);
110  }
111 
112  void insert(Event* ev, int64_t startDelay = -1) {
113  futex_lock(&qLock);
114  uint64_t curPhase = zinfo->numPhases;
115  uint64_t eventPhase = (startDelay == -1)? (curPhase + ev->getPeriod()) : (curPhase + startDelay);
116  assert(eventPhase >= curPhase);
117  evMap.insert(std::pair<uint64_t, Event*>(eventPhase, ev));
118  futex_unlock(&qLock);
119  }
120 };
