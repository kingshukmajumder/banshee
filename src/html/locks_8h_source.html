102 static inline void ticket_lock(volatile uint32_t* lock) {
103  /* Technically, we want to do this, but I'm guessing the 64-bit
104  * datapath is not very well optimized for 16-bit xadd...
105  * volatile uint16_t* low = ((volatile uint16_t*) lock) + 1;
106  * uint32_t ticket = atomic_fetchadd_16(low, 1);
107  */
108  uint32_t val, hi, newLo;
109  while (true) {
110  val = *lock;
111  hi = val & (TICKET_MASK << 16);
112  newLo = (val + 1) & TICKET_MASK;
113  if (__sync_bool_compare_and_swap(lock, val, (hi | newLo))) break;
114  }
115 
116  uint32_t ticket = val & TICKET_MASK;
117 
118  while ((((*lock) >> 16) & TICKET_MASK) != ticket) {
119 #if WITH_MWAIT
120  //Monitor / mwait
121  _mm_monitor((const void*)lock, 0, 0);
122 
123  //Must test again, might have intervening write BEFORE monitor (so we would get stuck in mwait)
124  if (*lock) {
125  _mm_mwait(0, 0);
126  }
127 #else
128  //If we don't have mwait, we can at least pause
129  _mm_pause();
130 #endif
131  }
132 }
