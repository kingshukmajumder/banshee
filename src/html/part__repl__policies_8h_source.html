181  void recordCandidate(uint32_t id) {
182  assert(candIdx < ways);
183  WayPartInfo* c = &array[id]; //candidate info
184  WayPartInfo* best = (bestId >= 0)? &array[bestId] : nullptr;
185  uint32_t way = candIdx++;
186  //In test mode, this works as LRU
187  if (testMode || wayPartIndex[way] == incomingLinePart) { //this is a way we can fill
188  if (best == nullptr) {
189  bestId = id;
190  } else {
191  //NOTE: This is actually not feasible without tagging. But what IS feasible is to stop updating the LRU position on new fills. We could kill this, and profile the differences.
192  if ( testMode || (c->p == incomingLinePart && best->p == incomingLinePart) ) {
193  if (c->ts < best->ts) bestId = id;
194  } else if (c->p == incomingLinePart && best->p != incomingLinePart) {
195  //c wins
196  } else if (c->p != incomingLinePart && best->p == incomingLinePart) {
197  //c loses
198  bestId = id;
199  } else { //none in our partition, this should be transient but at least enforce LRU
200  if (c->ts < best->ts) bestId = id;
201  }
202  }
203  }

459  uint32_t getBestCandidate() {
460  assert(candIdx > 0);
461  assert(candIdx <= assoc);
462 
463  //Demote all lines below their setpoints
464  for (uint32_t i = 0; i < candIdx; i++) {
465  LineInfo* e = &array[candList[i]];
466  if (e->ts == 0) continue; //empty, bypass
467 
468  uint32_t p = e->p;
469  if (p == partitions) continue; //bypass unmanaged region entries
470 
471  uint32_t size = partInfo[p].size;
472 
473  if (size <= partInfo[p].targetSize) continue; //bypass partitions below target
474 
475 #if VANTAGE_8BIT_BTS
476  //Must do mod 256 arithmetic. This will do generally worse because of wrap-arounds, but wrapping around is pretty rare
477  //TODO: Doing things this way, we can profile the difference between this and using larger coarse-grain timestamps
478  if (((partInfo[p].curBts - e->bts) % 256) /*8-bit distance to current TS*/ >= ((partInfo[p].curBts - partInfo[p].setpointBts) % 256)) {
479 #else
480  if (e->bts <= partInfo[p].setpointBts) {
481 #endif
482  // Demote!
483  // Out of p
484  partInfo[p].profDemotions.inc();
485  partInfo[p].size--;
486 
487  // Into unmanaged
488  e->p = partitions;
489  partInfo[partitions].size++;
490 
491  partInfo[p].curIntervalDems++;
492 
493  //Note extended size and op not affected
494  }
495 
496  partInfo[p].curIntervalCands++;
497 
498  // See if we need interval change
499  if (/*partInfo[p].curIntervalDems >= 16 || partInfo[p].curIntervalIns >= 16 ||*/ partInfo[p].curIntervalCands >= 256) {
500  double maxSz = partInfo[p].targetSize*(1.0 + partSlack);
501  double curSz = partInfo[p].size;
502  double aperture = 0.0;
503 
504  // Feedback-based aperture control
505  // TODO: Copy over the demotion thresholds lookup table code from the ISCA paper code, or quantize this.
506  // This is doing finer-grain demotions, but requires a bit more math.
507  if (curSz >= maxSz) {
508  aperture = maxAperture;
509  } else {
510  double slope = (maxAperture)/(maxSz - partInfo[p].targetSize);
511  assert(slope > 0.0);
512  aperture = slope*(curSz - partInfo[p].targetSize);
513  }
514 
515  if (aperture > 0.0) {
516 /*
517  info ("part %d setpoint adjust, curSz %f tgtSz %ld maxSz %f aperture %f curBts %ld setpointBts %ld interval cands %d ins %d dems %d cpt %f",
518  p, curSz, partInfo[p].targetSize, maxSz, aperture, partInfo[p].curBts, partInfo[p].setpointBts, partInfo[p].curIntervalCands,\
519  partInfo[p].curIntervalIns, partInfo[p].curIntervalDems, partInfo[p].curIntervalCands*aperture);
520 */
521 
522  int32_t shrink = partInfo[p].curIntervalDems;
523  if (shrink < aperture*partInfo[p].curIntervalCands) {
524  //info ("increasing setpoint");
525  if (partInfo[p].setpointBts < partInfo[p].curBts) partInfo[p].setpointBts++;
526  } else if (shrink > aperture*partInfo[p].curIntervalCands) {
527  //info ("decreasing setpoint");
528 #if VANTAGE_8BIT_BTS
529  //Never get the setpoint to go 256 positions behind the current timestamp
530  if ((partInfo[p].curBts - partInfo[p].setpointBts) < 255) partInfo[p].setpointBts--;
531 #else
532  if (partInfo[p].setpointBts > 0) partInfo[p].setpointBts--;
533 #endif
534  } else {
535  //info ("keeping setpoint");
536  }
537  }
538 
539  //info("part %d post setpointBts %ld", p, partInfo[p].setpointBts);
540 
541  partInfo[p].curIntervalCands = 0;
542  partInfo[p].curIntervalIns = 0;
543  partInfo[p].curIntervalDems = 0;
544  partInfo[p].setpointAdjs++;
545  }
546  } //for
547 
548  //Get best candidate for eviction
549  int32_t bestId = candList[0];
550 
551  for (uint32_t i = 0; i < candIdx; i++) { //note we include 0; 0 compares with itself, see shortcut to understand why
552  uint32_t id = candList[i];
553  LineInfo* e = &array[id];
554  LineInfo* best = &array[bestId];
555 
556  if (e->ts == 0) {
557  //shortcut for empty positions
558  bestId = id;
559  break;
560  }
561 
562  uint32_t p = e->p;
563 
564  if (p == partitions && best->p != partitions) { //prioritize umgd
565  bestId = id;
566  } else if (p == partitions && best->p == partitions) {
567  if (e->ts < best->ts) bestId = id;
568  } else if (p != partitions && best->p == partitions) {
569  //best wins, prioritize unmanaged
570  } else {
571  assert(p != partitions && best->p != partitions);
572  //Just do LRU; with correctly-sized partitions, this is VERY rare
573  //NOTE: If we were to study really small unmanaged regions, we can always get fancier and prioritize by aperture, bts, etc.
574  if (e->ts < best->ts) bestId = id;
575  }
576  }
577  assert(bestId >= 0 && (uint32_t)bestId < totalSize);
578  return bestId;

592  uint32_t s[partitions];
593  uint32_t usedSize = 0;
594  uint32_t linesToTakeAway = 0;
595  for (uint32_t p = 0; p < partitions; p++) {
596  s[p] = totalSize*sizes[p]/partGranularity;
597 #if UMON_INFO
598  info("part %d, %ld -> %d lines (now it's %ld lines) [cur %ld/%ld set %ld/%ld setAdjs %ld]", p, partInfo[p].targetSize, s[p],
599  partInfo[p].size, partInfo[p].curBts, partInfo[p].curBts % 256, partInfo[p].setpointBts, partInfo[p].setpointBts % 256, partInfo[p].setpointAdjs);
600 #endif
601  if (smoothTransients) {
602  partInfo[p].longTermTargetSize = s[p];
603  if (s[p] > partInfo[p].targetSize) { //growing
604  uint32_t newTarget = MAX(partInfo[p].targetSize, MIN(partInfo[p].longTermTargetSize, partInfo[p].size+1)); //always in [target,longTermTarget]
605  linesToTakeAway += newTarget - partInfo[p].targetSize;
606  partInfo[p].targetSize = newTarget;
607  }
608  } else {
609  partInfo[p].targetSize = s[p];
610  partInfo[p].longTermTargetSize = s[p];
611  }
612  usedSize += s[p];
613  }
614 
615  while (linesToTakeAway--) takeOneLine();
616 #if UMON_INFO
617  info("%d lines assigned, %d unmanaged", usedSize, totalSize - usedSize);
618 #endif
619  }

621  void takeOneLine() {
622  assert(smoothTransients);
623  uint32_t linesLeft = 0;
624  //NOTE: This is a fairly inefficient implementation, but we can do it cheaply in hardware
625  //Take away proportionally to difference between actual and long-term target
626  for (uint32_t p = 0; p < partitions; p++) {
627  int32_t left = partInfo[p].targetSize - partInfo[p].longTermTargetSize;
628  linesLeft += MAX(left, 0);
629  }
630  assert(linesLeft > 0);
631  uint32_t l = rng.randInt(linesLeft-1); //[0, linesLeft-1]
632  uint32_t curLines = 0;
633  for (uint32_t p = 0; p < partitions; p++) {
634  int32_t left = partInfo[p].targetSize - partInfo[p].longTermTargetSize;
635  curLines += MAX(left, 0);
636  if (left > 0 && l < curLines) {
637  partInfo[p].targetSize--;
638  return;
639  }
640  }
641  panic("Could not find any partition to take away space from???");
642  }
